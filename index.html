@register.simple_tag
def list_projects():
    return Project.objects.values('ID_project').distinct()[1:]

(en html) {% list_projects as IP_list %} para iterar los datos, ignora la paginación. Etiqueta de datos. La función es mostrar datos adicionales a los que envía un queryset de una View. GET DE UNA VIEW CLASS

Método get_queryset(self) petición de vista dentro de ella misma. Podemos capturar del objeto request las variables que pasamos como parámetros, en este caso en concreto linkeamos de la siguiente forma href=”?IP= value”, Django lo interpreta como una petición query trim. 

def get_queryset(self):
   IP = self.request.GET.get('IP')
   if IP:
       return Project.objects.filter(ID_project = IP)
   if IP==None:
       return Project.objects.all()[1:]
Capturamos el valor con el nombre correspondiente que aparece en href. Y con condicionales mencionamos la query específica a devolver

PETICIONES HTTP

Tener en cuenta que las peticiones http son inmutables, por tanto para poder editar sus variables es necesario deshabilitar este parámetro que los configura para poder efectuar cambios. 

class ProjectUpdateView(generic.UpdateView):
    model = Project
    fields = '__all__'
    template_name = 'Proyecto1/project_update_form.html'

    def post(self, request, *args, **kwargs):
        self.object = self.get_object()

        if self.request.POST.get('_addanother'):
            print(self.object.id)
            new_PI = Project()
            self.object.id = new_PI.id

            kwargs['pk'] = str(new_PI.id)
            request.POST._mutable = True
            request.POST['id'] = str(new_PI.id)
            request.POST._mutable = False
            print("\n\n\nCreando un objeto nuevo:\n")
            #kwargs.update({'pk': None})
            print(kwargs)
            print(self.object.id)
            print(request.POST)
            print(args)

        return super().post(request, *args, kwargs)
